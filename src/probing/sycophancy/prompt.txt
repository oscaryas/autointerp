## Step 5 — Top-k Activation Steering (per model)

Verify the probe direction is causally linked to sycophancy by steering at the top-16 MHA heads.

```python
# Select top-k heads by accuracy
k_heads = 16
top_heads = sorted(mha_acc, key=mha_acc.get, reverse=True)[:k_heads]

# Load saved weights and projection stds (written by save_probe_results)
weights_ckpt   = torch.load(f"final_probe_{model_name_safe}/mha_probe_weights.pth")
proj_stds_ckpt = torch.load(f"final_probe_{model_name_safe}/mha_projection_stds.pt")

head_dim = config["head_dim"]
n_heads  = config["n_heads"]
mha_hook = config["mha_hook"]

def build_steering_hooks(top_heads, weights_ckpt, proj_stds_ckpt, scale):
    """Build one pre-hook per unique layer in top_heads."""
    from collections import defaultdict
    per_layer = defaultdict(list)   # layer -> [(head, direction, proj_std)]
    for (layer, head) in top_heads:
        w = weights_ckpt[(layer, head)]["linear.weight"][0].numpy()
        direction = w / (np.linalg.norm(w) + 1e-8)
        proj_std  = proj_stds_ckpt[(layer, head)]
        per_layer[layer].append((head, direction, proj_std))

    hooks = {}
    for layer, head_list in per_layer.items():
        def make_hook(hlist, hd, sc):
            def hook(module, inp):
                x = inp[0].clone()
                for (h, d, s) in hlist:
                    start = h * hd
                    end   = start + hd
                    vec   = torch.tensor(d * s * sc, dtype=x.dtype, device=x.device)
                    x[:, :, start:end] += vec
                return (x,)
            return hook
        hooks[layer] = make_hook(head_list, head_dim, scale)
    return hooks

def register_steering(model, top_heads, weights_ckpt, proj_stds_ckpt, scale):
    hooks = build_steering_hooks(top_heads, weights_ckpt, proj_stds_ckpt, scale)
    handles = []
    for name, module in model.named_modules():
        if name.endswith(mha_hook):
            try:
                from sycophancy_model_registry import _extract_layer_idx
                li = _extract_layer_idx(name)
            except ValueError:
                continue
            if li in hooks:
                handles.append(module.register_forward_pre_hook(hooks[li]))
    return handles

# Measure sycophancy_rate at baseline, steered away (scale=-5), steered toward (scale=+5)
# Use 20 held-out examples from behavioral_labels
from sycophancy_data import detect_capitulation, generate_response, compute_sycophancy_rate
from sycophancy_probes import bootstrap_delta_ci

held_out = []
with open(f"behavioral_labels_{model_name_safe}.jsonl") as f:
    for i, line in enumerate(f):
        if i >= 20: break
        held_out.append(json.loads(line))

def run_inference(model, tokenizer, examples, scale=None):
    """Run inference with optional steering hooks. Returns binary label array."""
    handles = []
    if scale is not None:
        handles = register_steering(model, top_heads, weights_ckpt, proj_stds_ckpt, scale)
    labels = []
    for ex in examples:
        resp = generate_response(model, tokenizer, ex["text"])
        labels.append(detect_capitulation(ex["initial_answer"], ex["alternative"], resp))
    for h in handles:
        h.remove()
    return np.array(labels)

baseline_labels = run_inference(model, tokenizer, held_out)
labels_away     = run_inference(model, tokenizer, held_out, scale=-5.0)
labels_toward   = run_inference(model, tokenizer, held_out, scale=+5.0)

steered_away_rate   = float(labels_away.mean())
steered_toward_rate = float(labels_toward.mean())

delta_away_ci   = bootstrap_delta_ci(baseline_labels.astype(float), labels_away.astype(float))
delta_toward_ci = bootstrap_delta_ci(baseline_labels.astype(float), labels_toward.astype(float))

steering_results = {
    "model": model_name,
    "k_heads": k_heads,
    "scale": 5.0,
    "n_examples": len(held_out),
    "baseline_sycophancy_rate": float(baseline_labels.mean()),
    "steered_away_rate": steered_away_rate,
    "steered_toward_rate": steered_toward_rate,
    "delta_away": steered_away_rate - float(baseline_labels.mean()),
    "delta_away_ci_95": list(delta_away_ci),
    "delta_toward": steered_toward_rate - float(baseline_labels.mean()),
    "delta_toward_ci_95": list(delta_toward_ci),
    "top_heads": [[l, h] for l, h in top_heads],
}
with open(f"steering_results_{model_name_safe}.json", "w") as f:
    json.dump(steering_results, f, indent=2)
```

Expected: `steered_toward_rate > baseline > steered_away_rate`. The CI on each delta should exclude zero to be considered significant.

## Step 6 — Aggregate Attention Score Heatmap (per model)

This answers a **different question** from probe accuracy: what do the top-k sycophancy heads actually attend to?

Run on 20 held-out examples with `output_attentions=True`. Use `turn_boundaries` from `behavioral_labels` to segment each prompt into four dialogue regions.

```python
# Load held-out examples with turn_boundaries
held_out_with_bounds = []
with open(f"behavioral_labels_{model_name_safe}.jsonl") as f:
    for i, line in enumerate(f):
        if i >= 20: break
        ex = json.loads(line)
        if ex.get("turn_boundaries"):
            held_out_with_bounds.append(ex)

# Collect attention weights for top-k heads at the final answer position
region_labels = ["Question", "First answer", "Disagreement", "Final answer"]
attention_matrix = np.zeros((len(top_heads), 4))  # (top_k, n_regions)

for ex in held_out_with_bounds:
    inputs = tokenizer(ex["text"], return_tensors="pt", truncation=True, max_length=1024)
    if torch.cuda.is_available():
        inputs = {k: v.to(model.device) for k, v in inputs.items()}
    with torch.no_grad():
        outputs = model(**inputs, output_attentions=True)

    # outputs.attentions: tuple of (1, n_heads, seq_len, seq_len) per layer
    bounds = ex["turn_boundaries"]
    regions = [(0, bounds[0]), (bounds[0], bounds[1]), (bounds[1], bounds[2]), (bounds[2], bounds[3])]
    seq_len = inputs["input_ids"].shape[1]
    final_pos = min(bounds[3] - 1, seq_len - 1)

    for ki, (layer, head) in enumerate(top_heads):
        if layer < len(outputs.attentions):
            attn = outputs.attentions[layer][0, head, final_pos, :].cpu().numpy()
            for ri, (r_start, r_end) in enumerate(regions):
                r_end = min(r_end, seq_len)
                attention_matrix[ki, ri] += float(attn[r_start:r_end].sum())

if held_out_with_bounds:
    attention_matrix /= len(held_out_with_bounds)

# Plot: rows=top-k heads, columns=dialogue regions
fig, ax = plt.subplots(figsize=(8, max(6, len(top_heads) // 2)))
im = ax.imshow(attention_matrix, aspect="auto", cmap="Blues", vmin=0)
plt.colorbar(im, ax=ax, label="Mean attention mass")
ax.set_xticks(range(4))
ax.set_xticklabels(region_labels, rotation=30, ha="right")
ax.set_yticks(range(len(top_heads)))
ax.set_yticklabels([f"L{l}H{h}" for l, h in top_heads], fontsize=7)
ax.set_title(f"{model_name}\nTop-{k_heads} sycophancy heads — attention to dialogue regions")
plt.tight_layout()
plt.savefig(f"aggregate_attention_score_heatmap_{model_name_safe}.png", dpi=150)
plt.close()

# Bootstrap CI on attention mass per region across examples
# attention_per_example[i, region] = total attention mass for example i
attention_per_example = np.zeros((len(held_out_with_bounds), 4))
for ex_i, ex in enumerate(held_out_with_bounds):
    inputs = tokenizer(ex["text"], return_tensors="pt", truncation=True, max_length=1024)
    if torch.cuda.is_available():
        inputs = {k: v.to(model.device) for k, v in inputs.items()}
    with torch.no_grad():
        outputs = model(**inputs, output_attentions=True)
    bounds = ex["turn_boundaries"]
    seq_len = inputs["input_ids"].shape[1]
    final_pos = min(bounds[3] - 1, seq_len - 1)
    regions = [(0, bounds[0]), (bounds[0], bounds[1]), (bounds[1], bounds[2]), (bounds[2], min(bounds[3], seq_len))]
    for ki, (layer, head) in enumerate(top_heads):
        if layer < len(outputs.attentions):
            attn = outputs.attentions[layer][0, head, final_pos, :].cpu().numpy()
            for ri, (r_start, r_end) in enumerate(regions):
                attention_per_example[ex_i, ri] += float(attn[r_start:r_end].sum())

n_ex = len(held_out_with_bounds)
attention_mean = attention_per_example.mean(axis=0) if n_ex > 0 else np.zeros(4)
attention_ci = []
for ri in range(4):
    if n_ex >= 2:
        from sycophancy_probes import bootstrap_ci as bci
        lo, hi = bci(attention_per_example[:, ri])
    else:
        lo, hi = 0.0, 0.0
    attention_ci.append([round(lo, 4), round(hi, 4)])

with open(f"aggregate_attention_score_{model_name_safe}.json", "w") as f:
    json.dump({
        "model": model_name,
        "n_examples": n_ex,
        "top_heads": [[l, h] for l, h in top_heads],
        "region_labels": region_labels,
        "attention_mean": attention_mean.tolist(),
        "attention_ci_95": attention_ci,
    }, f, indent=2)
```

## Step 7 — Compute metrics.json (per model)

```python
# Compute sycophancy_rate from behavioral labels
examples_for_rate = []
with open(f"behavioral_labels_{model_name_safe}.jsonl") as f:
    for line in f:
        examples_for_rate.append(json.loads(line))

first_correct_count  = sum(1 for e in examples_for_rate if e.get("first_correct"))
second_correct_count = sum(1 for e in examples_for_rate if e.get("second_correct"))
correct_to_incorrect = sum(1 for e in examples_for_rate if e.get("first_correct") and not e.get("second_correct"))
n = len(examples_for_rate)
sycophancy_rate = correct_to_incorrect / first_correct_count if first_correct_count else 0.0

with open(f"final_probe_{model_name_safe}/probe_metadata.json") as f:
    meta = json.load(f)

from sycophancy_probes import bootstrap_ci as bci

# Sycophancy rate bootstrap CI (binary 0/1 labels from behavioral data)
syco_labels = np.array([
    1 if e.get("first_correct") and not e.get("second_correct") else 0
    for e in examples_for_rate
], dtype=float)
syco_rate_ci = list(bci(syco_labels)) if len(syco_labels) >= 2 else [0.0, 0.0]

metrics = {
    "model_name": model_name,
    "mha_best_accuracy": meta["mha_best_accuracy"],
    "mha_best_ci_95": meta.get("mha_best_ci"),
    "mlp_best_accuracy": meta["mlp_best_accuracy"],
    "mlp_best_ci_95": meta.get("mlp_best_ci"),
    "residual_best_accuracy": meta["residual_best_accuracy"],
    "residual_best_ci_95": meta.get("residual_best_ci"),
    "first_answer_accuracy": first_correct_count / n if n else 0.0,
    "second_answer_accuracy": second_correct_count / n if n else 0.0,
    "sycophancy_rate": round(sycophancy_rate, 4),
    "sycophancy_rate_ci_95": [round(syco_rate_ci[0], 4), round(syco_rate_ci[1], 4)],
    "n_first_correct": first_correct_count,
    "n_correct_to_incorrect": correct_to_incorrect,
    # Aggregate attention score (mean mass per dialogue region for top-k heads)
    "attention_region_labels": region_labels,
    "attention_mean_per_region": [round(float(v), 4) for v in attention_mean.tolist()],
    "attention_ci_95_per_region": attention_ci,
    "memory_policy": "conservative",
    "paper_verified": False,
}
with open(f"metrics_{model_name_safe}.json", "w") as f:
    json.dump(metrics, f, indent=2)
```

## Step 10 — Cross-Model Summary (after loop)

Run after all models complete. Use relative layer depth — do NOT average raw head indices across architectures.

```python
from scipy.interpolate import interp1d

N_BINS = 20
all_depth_curves = []

for entry in all_model_results:
    mha_acc_e = entry["mha_accuracy"]
    n_layers_e = entry["config"]["n_layers"]
    # Max accuracy per layer (across all heads at that layer)
    per_layer = {}
    for (layer, head), acc in mha_acc_e.items():
        per_layer[layer] = max(per_layer.get(layer, 0.0), acc)
    if not per_layer:
        continue
    x_old = np.array(sorted(per_layer.keys())) / max(n_layers_e - 1, 1)
    y_old = np.array([per_layer[k] for k in sorted(per_layer.keys())])
    x_new = np.linspace(0, 1, N_BINS)
    curve = interp1d(x_old, y_old, bounds_error=False, fill_value="extrapolate")(x_new)
    all_depth_curves.append(curve * 100)

if len(all_depth_curves) >= 2:
    depth_matrix = np.stack(all_depth_curves)   # (n_models, N_BINS)
    fig, ax = plt.subplots(figsize=(12, 4))
    for i, entry in enumerate(all_model_results):
        ax.plot(np.linspace(0, 100, N_BINS), depth_matrix[i],
                label=entry["model_name"].split("/")[-1], alpha=0.8)
    ax.set_xlabel("Relative layer depth (%)")
    ax.set_ylabel("Max MHA probe accuracy (%)")
    ax.set_title("MHA sycophancy probe accuracy by relative layer depth")
    ax.legend(fontsize=7)
    plt.tight_layout()
    plt.savefig("mha_probe_accuracy_summary_by_depth.png", dpi=150)
    plt.close()
```
