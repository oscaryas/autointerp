## Step 3 — Extract Activations and Train Probes

```python
import json
import numpy as np
from sycophancy_probes import extract_and_train_all, save_probe_results
from gpu_memory import get_memory_diagnostics, safe_cleanup

# Load labels
texts, labels = [], []
with open("behavioral_labels.jsonl") as f:
    for line in f:
        item = json.loads(line)
        texts.append(item["text"])
        labels.append(item["label"])
labels = np.array(labels)

# Extract activations and train all three probe types
results = extract_and_train_all(model, tokenizer, texts, labels, config)

# Save probe results and memory diagnostics
probe_meta = save_probe_results(results, "final_probe", model_name="{model}")

mem_diag = get_memory_diagnostics("{model}", dtype, batch_size=1)
with open("final_probe/memory_diagnostics.json", "w") as f:
    json.dump(mem_diag, f, indent=2)

# Mandatory: clean up model before loading the next one
safe_cleanup(model, tokenizer, job_dir=".")
```
